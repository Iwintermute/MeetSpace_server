cmake_minimum_required(VERSION 3.20)
project(EDS_serverNew LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Collect all source files recursively
file(GLOB_RECURSE SOURCES CONFIGURE_DEPENDS "*.cpp")
file(GLOB_RECURSE HEADERS CONFIGURE_DEPENDS "*.h" "*.hpp")
set(MEDIASOUP_BRIDGE_SOURCES
  "${CMAKE_CURRENT_SOURCE_DIR}/Bridge/Mediasoup/runtime/MediasoupDebugConfig.cpp"
  "${CMAKE_CURRENT_SOURCE_DIR}/Bridge/Mediasoup/service/MediasoupTransportService.cpp"
  "${CMAKE_CURRENT_SOURCE_DIR}/Bridge/Mediasoup/service/MediasoupSfuEngine.cpp"
)
set(MEDIASOUP_BRIDGE_HEADERS
  "${CMAKE_CURRENT_SOURCE_DIR}/Bridge/Mediasoup/runtime/MediasoupDebugConfig.h"
  "${CMAKE_CURRENT_SOURCE_DIR}/Bridge/Mediasoup/service/IMediaEngine.h"
  "${CMAKE_CURRENT_SOURCE_DIR}/Bridge/Mediasoup/service/IMediaTransportService.h"
  "${CMAKE_CURRENT_SOURCE_DIR}/Bridge/Mediasoup/service/MediaTransportTypes.h"
  "${CMAKE_CURRENT_SOURCE_DIR}/Bridge/Mediasoup/service/MediasoupTransportService.h"
  "${CMAKE_CURRENT_SOURCE_DIR}/Bridge/Mediasoup/service/MediasoupSfuEngine.h"
)
set(MEDIASOUP_LEGACY_RUNTIME_SOURCES
  "${CMAKE_CURRENT_SOURCE_DIR}/Bridge/Mediasoup/runtime/MediasoupActions.cpp"
  "${CMAKE_CURRENT_SOURCE_DIR}/Bridge/Mediasoup/runtime/MediasoupFeatureManager.cpp"
  "${CMAKE_CURRENT_SOURCE_DIR}/Bridge/Mediasoup/runtime/MediasoupRtcBridge.cpp"
  "${CMAKE_CURRENT_SOURCE_DIR}/Bridge/Mediasoup/runtime/MediasoupSignalingAgent.cpp"
  "${CMAKE_CURRENT_SOURCE_DIR}/Bridge/Mediasoup/runtime/MediasoupStateStore.cpp"
)

set(MEDIASOUP_LEGACY_RUNTIME_HEADERS
  "${CMAKE_CURRENT_SOURCE_DIR}/Bridge/Mediasoup/runtime/MediasoupActions.h"
  "${CMAKE_CURRENT_SOURCE_DIR}/Bridge/Mediasoup/runtime/MediasoupFeatureManager.h"
  "${CMAKE_CURRENT_SOURCE_DIR}/Bridge/Mediasoup/runtime/MediasoupRtcBridge.h"
  "${CMAKE_CURRENT_SOURCE_DIR}/Bridge/Mediasoup/runtime/MediasoupSignalingAgent.h"
  "${CMAKE_CURRENT_SOURCE_DIR}/Bridge/Mediasoup/runtime/MediasoupStateStore.h"
)
list(REMOVE_ITEM SOURCES ${MEDIASOUP_BRIDGE_SOURCES})
list(REMOVE_ITEM HEADERS ${MEDIASOUP_BRIDGE_HEADERS})
list(REMOVE_ITEM SOURCES ${MEDIASOUP_LEGACY_RUNTIME_SOURCES})
list(REMOVE_ITEM HEADERS ${MEDIASOUP_LEGACY_RUNTIME_HEADERS})
list(REMOVE_ITEM SOURCES "${CMAKE_CURRENT_SOURCE_DIR}/main.cpp")

set(TARGET_FILE "eds_server_new_arch")
set(MEDIASOUP_BRIDGE_TARGET "eds_mediasoup_bridge")

add_library(${MEDIASOUP_BRIDGE_TARGET} STATIC ${MEDIASOUP_BRIDGE_SOURCES} ${MEDIASOUP_BRIDGE_HEADERS})

# New architecture target (rewrite in progress)
add_library(${TARGET_FILE} STATIC ${SOURCES} ${HEADERS})

set(_eduspace_word_generator_cpp "")
if(DEFINED EDUSPACE_RESOLVED_CORE_DIR AND EXISTS "${EDUSPACE_RESOLVED_CORE_DIR}/utils/WordGenerator.cpp")
  set(_eduspace_word_generator_cpp "${EDUSPACE_RESOLVED_CORE_DIR}/utils/WordGenerator.cpp")
elseif(EXISTS "${CMAKE_SOURCE_DIR}/Core/utils/WordGenerator.cpp")
  set(_eduspace_word_generator_cpp "${CMAKE_SOURCE_DIR}/Core/utils/WordGenerator.cpp")
endif()
if(_eduspace_word_generator_cpp)
  target_sources(${TARGET_FILE} PRIVATE "${_eduspace_word_generator_cpp}")
endif()


# Include directories
target_include_directories(${MEDIASOUP_BRIDGE_TARGET} PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}
)
target_include_directories(${TARGET_FILE} PRIVATE 
    ${CMAKE_CURRENT_SOURCE_DIR}
)

if(NOT TARGET eduspace_core)
  message(FATAL_ERROR "Target eduspace_core is required. Ensure Core is configured in the root CMakeLists.")
endif()
target_link_libraries(${MEDIASOUP_BRIDGE_TARGET} PRIVATE
    eduspace_core
)
target_link_libraries(${TARGET_FILE} PRIVATE
    eduspace_core
    ${MEDIASOUP_BRIDGE_TARGET}
)

find_package(OpenSSL REQUIRED)
target_link_libraries(${MEDIASOUP_BRIDGE_TARGET} PRIVATE
  OpenSSL::SSL
  OpenSSL::Crypto
)
target_link_libraries(${TARGET_FILE} PRIVATE
  OpenSSL::SSL
  OpenSSL::Crypto
)

set(_eduspace_vcpkg_include_candidates "")
if(DEFINED VCPKG_INSTALLED_DIR AND DEFINED VCPKG_TARGET_TRIPLET)
  list(APPEND _eduspace_vcpkg_include_candidates
    "${VCPKG_INSTALLED_DIR}/${VCPKG_TARGET_TRIPLET}/include"
  )
endif()
if(DEFINED ENV{VCPKG_ROOT})
  if(DEFINED VCPKG_TARGET_TRIPLET)
    list(APPEND _eduspace_vcpkg_include_candidates
      "$ENV{VCPKG_ROOT}/installed/${VCPKG_TARGET_TRIPLET}/include"
    )
  endif()
  list(APPEND _eduspace_vcpkg_include_candidates
    "$ENV{VCPKG_ROOT}/installed/x64-windows/include"
  )
endif()

set(_eduspace_boost_include "")
foreach(_candidate IN LISTS _eduspace_vcpkg_include_candidates)
  if(EXISTS "${_candidate}/boost/asio.hpp")
    set(_eduspace_boost_include "${_candidate}")
    break()
  endif()
endforeach()
if(_eduspace_boost_include)
  target_include_directories(${MEDIASOUP_BRIDGE_TARGET} PUBLIC "${_eduspace_boost_include}")
  target_compile_definitions(${MEDIASOUP_BRIDGE_TARGET} PUBLIC BOOST_ERROR_CODE_HEADER_ONLY)
  target_include_directories(${TARGET_FILE} PUBLIC "${_eduspace_boost_include}")
  target_compile_definitions(${TARGET_FILE} PUBLIC BOOST_ERROR_CODE_HEADER_ONLY)
else()
  message(WARNING "Boost headers were not auto-detected in vcpkg include paths. Mediasoup signaling transport may fail to compile.")
endif()

if(WIN32)
  target_compile_definitions(${MEDIASOUP_BRIDGE_TARGET} PUBLIC _WIN32_WINNT=0x0A00)
  target_link_libraries(${MEDIASOUP_BRIDGE_TARGET} PRIVATE ws2_32 mswsock)
  target_compile_definitions(${TARGET_FILE} PUBLIC _WIN32_WINNT=0x0A00)
  target_link_libraries(${TARGET_FILE} PRIVATE ws2_32 mswsock)
endif()

add_executable(eds_server_new_mediasoup_app main.cpp)
target_include_directories(eds_server_new_mediasoup_app PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}
)
target_link_libraries(eds_server_new_mediasoup_app PRIVATE
    ${TARGET_FILE}
    ${MEDIASOUP_BRIDGE_TARGET}
    eduspace_core
)

if(MSVC)
  target_link_options(eds_server_new_mediasoup_app PRIVATE
    "/WHOLEARCHIVE:eds_server_new_arch.lib"
  )
elseif(CMAKE_CXX_COMPILER_ID MATCHES "GNU|Clang")
  target_link_options(eds_server_new_mediasoup_app PRIVATE
    "-Wl,--whole-archive"
    "$<TARGET_FILE:${TARGET_FILE}>"
    "-Wl,--no-whole-archive"
  )
endif()

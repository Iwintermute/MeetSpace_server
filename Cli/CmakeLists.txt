cmake_minimum_required(VERSION 3.20)
project(Cli)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Collect all source files recursively
file(GLOB_RECURSE SOURCES "*.cpp")
file(GLOB_RECURSE HEADERS "*.h" "*.hpp")

set(TARGET_FILE "Cli")

# Create executable
add_executable(${TARGET_FILE} ${SOURCES} ${HEADERS})

# Include directories
target_include_directories(${TARGET_FILE} PRIVATE ${CMAKE_CURRENT_SOURCE_DIR})

if(NOT TARGET eduspace_core)
  message(FATAL_ERROR "Target eduspace_core is required. Ensure Core is configured in the root CMakeLists.")
endif()

find_package(OpenSSL REQUIRED)
find_package(LibDataChannel CONFIG REQUIRED)
find_package(portaudio CONFIG REQUIRED)
find_package(opus CONFIG REQUIRED)

  target_link_libraries(${TARGET_FILE} PRIVATE 
    eduspace_core
		LibDataChannel::LibDataChannel
		portaudio
		Opus::opus
		OpenSSL::SSL
		OpenSSL::Crypto
  )
